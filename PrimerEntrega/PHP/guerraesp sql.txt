CREATE DATABASE guerraesp;
USE guerraesp;

CREATE TABLE Usuarios(
    IDUsuario INT IDENTITY PRIMARY KEY,
    NomApe NVARCHAR(50) NOT NULL,
    Nickname NVARCHAR(50) NOT NULL,
    Contraseña NVARCHAR(512) NOT NULL,
    Correo NVARCHAR(100) UNIQUE NOT NULL,
    AceptacionTerminos BIT NOT NULL
);

CREATE TABLE Administradores(
    IDUsuario INT PRIMARY KEY,
    FOREIGN KEY (IDUsuario) REFERENCES Usuarios(IDUsuario)
);

CREATE TABLE Jugadores(
    IDUsuario INT PRIMARY KEY,
    FOREIGN KEY (IDUsuario) REFERENCES Usuarios(IDUsuario)
);

CREATE TABLE Partidas(
    IDPartida INT IDENTITY PRIMARY KEY,
    perdedor INT,
    Fecha_hora DATETIME DEFAULT GETDATE() NOT NULL,
    Estado NVARCHAR(10) CHECK(Estado IN ('en pausa', 'terminado', 'jugando')),
    ganador INT
);

CREATE TABLE Rondas(
    IDRonda INT IDENTITY PRIMARY KEY,
    ganador INT,
    NumeroRonda INT NOT NULL,
    IDPartida INT NOT NULL,
    FOREIGN KEY (IDPartida) REFERENCES Partidas(IDPartida)
);

CREATE TABLE Batallas(
    IDBatalla INT IDENTITY PRIMARY KEY,
    ganador INT NOT NULL,
    NumeroBatalla INT NOT NULL,
    IDRonda INT NOT NULL,
    FOREIGN KEY (IDRonda) REFERENCES Rondas(IDRonda)
);

CREATE TABLE Cartas(
    IDCarta INT IDENTITY PRIMARY KEY,
    Numero INT NOT NULL,
    Palo INT NOT NULL,
    Ruta_img NVARCHAR(100) NOT NULL,
    IDBatalla INT NOT NULL,
    FOREIGN KEY (IDBatalla) REFERENCES Batallas(IDBatalla)
);

CREATE TABLE Mazo(
    IDMazo INT IDENTITY PRIMARY KEY,
    IDUsuario INT NOT NULL,
    IDPartida INT NOT NULL,
    FOREIGN KEY (IDUsuario) REFERENCES Jugadores(IDUsuario),
    FOREIGN KEY (IDPartida) REFERENCES Partidas(IDPartida)
);

CREATE TABLE Participan(
    IDUsuario INT NOT NULL,
    IDPartida INT NOT NULL,
    PRIMARY KEY (IDUsuario, IDPartida),
    FOREIGN KEY (IDUsuario) REFERENCES Jugadores(IDUsuario),
    FOREIGN KEY (IDPartida) REFERENCES Partidas(IDPartida)
);

CREATE TABLE Juegan(
    IDUsuario INT NOT NULL,
    IDCarta INT NOT NULL,
    IDBatalla INT NOT NULL,
    PRIMARY KEY (IDUsuario, IDCarta, IDBatalla),
    FOREIGN KEY (IDUsuario) REFERENCES Jugadores(IDUsuario),
    FOREIGN KEY (IDCarta) REFERENCES Cartas(IDCarta),
    FOREIGN KEY (IDBatalla) REFERENCES Batallas(IDBatalla)
);

CREATE TABLE Tienen(
    IDMazo INT NOT NULL,
    IDCarta INT NOT NULL,
    PRIMARY KEY (IDMazo, IDCarta),
    FOREIGN KEY (IDMazo) REFERENCES Mazo(IDMazo),
    FOREIGN KEY (IDCarta) REFERENCES Cartas(IDCarta)
);

GO
CREATE PROCEDURE InsertarUsuarioPartida
    @IDUsario INT,
    @IDUsuario2 INT,
    @Estado NVARCHAR(10)

AS
BEGIN
    INSERT INTO Partidas (Estado)
    VALUES (@Estado);

    DECLARE @IDPartida INT = SCOPE_IDENTITY();

    INSERT INTO Participan (IDPartida, IDUsuario)
    VALUES (@IDPartida, @IDUsario);

    INSERT INTO Participan (IDPartida, IDUsuario)
    VALUES (@IDPartida, @IDUsuario2);

END
GO
GO
CREATE PROCEDURE CrearUsuario
    @NomApe NVARCHAR(50),
    @Nickname NVARCHAR(50),
    @Contraseña NVARCHAR(512),
    @Correo NVARCHAR(100),
    @AceptacionTerminos BIT,
    @Tipo VARCHAR(10)

AS
BEGIN
    INSERT INTO Usuarios (NomApe, Nickname, Contraseña, Correo, AceptacionTerminos)
    VALUES (@NomApe, @Nickname, @Contraseña, @Correo, @AceptacionTerminos);

    DECLARE @IDUsuario INT = SCOPE_IDENTITY();

    IF @Tipo = 'Admin'
    BEGIN
        INSERT INTO Administradores(IDUsuario)
        VALUES (@IDUsuario);
    END
    ELSE
    BEGIN
        INSERT INTO Jugadores(IDUsuario)
        VALUES (@IDUsuario);
    END

END
GO
